  let guess = Number(document.getElementById("guessInput").value);
  tries++;
  if (guess < secret) {
    showMessage("Too Low!");
  } else if (guess > secret) {
    showMessage("Too High!");
  } else {
    showMessage("Correct! Took " + tries + " tries.");
  }
}

function convert() {
  let val = parseFloat(document.getElementById("val").value);
  let unit = document.getElementById("unit").value;
  if (isNaN(val)) { showError("Enter a valid number"); return; }

  let inMeters = val * toMeters[unit];
  document.getElementById("km").textContent = (inMeters / 1000) + " km";
  document.getElementById("m").textContent  = inMeters + " m";
  document.getElementById("cm").textContent = (inMeters * 100) + " cm";
  document.getElementById("mm").textContent = (inMeters * 1000) + " mm";
}

Step 3: Hook both elements to the same function

<select onchange="convert()" id="unit">...</select>

function updateClock() {
  let now = new Date();
  let h = now.getHours();
  let m = String(now.getMinutes()).padStart(2, "0");
  let s = String(now.getSeconds()).padStart(2, "0");
  let hh = String(h).padStart(2, "0");

  document.getElementById("clock").textContent = hh + ":" + m + ":" + s;
  document.getElementById("greeting").textContent = getGreeting(h);
}

Step 2: Decide greeting based on hour

function getGreeting(h) {
  if (h >= 5 && h < 12)  return "Good Morning";
  if (h >= 12 && h < 18) return "Good Afternoon";
  if (h >= 18 && h < 22) return "Good Evening";
  return "Good Night";
}

Step 3: Start the clock (runs every 1 second)

updateClock();
setInterval(updateClock, 1000);

How to build it — step by step

The key pattern — flag + per-field errors

function validateForm() {
  let valid = true;

  let name = document.getElementById("name").value.trim();
  if (name.length < 3) {
    showError("nameErr", "Name must be at least 3 characters");
    valid = false;
  } else { clearError("nameErr"); }

  let email = document.getElementById("email").value;
  if (!email.includes("@")) {
    showError("emailErr", "Email must contain @");
    valid = false;
  } else { clearError("emailErr"); }

  let age = Number(document.getElementById("age").value);
  if (age < 18 || age > 60) {
    showError("ageErr", "Age must be between 18 and 60");
    valid = false;
  } else { clearError("ageErr"); }

  let agreed = document.getElementById("terms").checked;
  if (!agreed) {
    showError("termsErr", "You must agree to the terms");
    valid = false;
  } else { clearError("termsErr"); }

  if (valid) showSuccess("Form Submitted Successfully!");
}
